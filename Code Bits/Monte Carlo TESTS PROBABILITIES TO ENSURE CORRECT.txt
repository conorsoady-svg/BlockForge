// Test that weightedRandomShape respects 40/35/25 using the REAL game tiers
(function testWeightedPicker(samples = 50000) {
  const counts = { easy: 0, medium: 0, hard: 0 };

  for (let i = 0; i < samples; i++) {
    const s = window.weightedRandomShape();
    const t = window.tierOf(s);   // <-- uses the game's mapping from shared.js
    counts[t] = (counts[t] || 0) + 1;
  }

  const pct = Object.fromEntries(
    Object.entries(counts).map(([k, v]) => [
      k,
      (v / samples * 100).toFixed(2) + '%'
    ])
  );

  console.log('Samples:', samples);
  console.table({ counts, pct });
})(50000);
